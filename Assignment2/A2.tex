\begin{problem}
    (8 marks) Add the following, indicate if there is a type of overflow or carry over for a 8 bit binary. 
    \begin{enumerate}[i]
        \item $12 + 10$
        \item $0100 1100 + 0011 1110$
        \item $1001 0000 + 1111 1111$
        \item $4 + -13$
    \end{enumerate}
\end{problem}

        \item $9 - 10$
        \item $0000 1010 - 1111 1111$
        \item $4 - 13$
    \end{enumerate}
